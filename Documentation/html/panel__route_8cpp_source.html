151 void PanelRoute::parseMessage(QByteArray data)
152 {
153  QString incomingMessage = data;
154  QString errorMessage;
155  QThread wait;
156 // qDebug() << "Message Type: " << nextParseState;
157 // qDebug() << "New incoming message: " << incomingMessage << " Message size: " << data.size();
158 
159  switch(nextParseState)
160  {
161  // *********************************************************************
162  // Checks to see if there is a route stored on the flight control board.
163  // *********************************************************************
164  case ONBOARD_QUERY_ROUTE:
165  waypointCount = incomingMessage.toInt();
166  if (waypointCount)
167  {
168  route->clearWaypoints();
169  updateRouteTable();
170  waypointIndex = 0;
171  sendMessage("o0;");
172  nextParseState = ONBOARD_REQUEST_WAYPOINT;
173  }
174  else // no route onboard flight control board, request position data
175  {
176  emit panelStatus("Route not detected on flight control board, loading last loaded route from file.");
177  startPositionRequest();
178  }
179  break;
180  // *********************************************************************
181  // Loads route from board and verifies if waypoint message has correct
182  // number of parameters.
183  // *********************************************************************
184  case ONBOARD_REQUEST_WAYPOINT:
185  {
186  QStringList waypoint = incomingMessage.split(',');
187  // Verify waypoint message by checking number of parameters
188  if (waypoint.size() == 4)
189  {
190  qreal lat = waypoint[1].toDouble()/1.0E7;
191  qreal lon = waypoint[2].toDouble()/1.0E7;
192  qreal alt = waypoint[3].toDouble();
193  route->addWaypoint(lon, lat, alt, 0.0, GeoDataCoordinates::Degree);
194  retryMessage = 0;
195 
196  waypointIndex++;
197  if (waypointIndex < waypointCount) // Setup command for next waypoint request
198  {
199  sendMessage("o" + QString::number(waypointIndex) + ";");
200  nextParseState = ONBOARD_REQUEST_WAYPOINT;
201  }
202  else // finished route retrieveal from flight control board, update route table
203  {
204  updateRouteTable();
205  emit panelStatus("Loaded route from flight control board successfully!");
206  startPositionRequest();
207  }
208  }
209  else // Incorrect number of parameters received, retry waypoint request
210  {
211  wait.msleep(200);
212  retryMessage++;
213  if (retryMessage<MAX_MSG_RETRY)
214  {
215  sendMessage( "o" + QString::number(waypointIndex) + ";");
216  nextParseState = ONBOARD_REQUEST_WAYPOINT;
217  }
218  else
219  {
220  errorMessage = "Unable to retrieve stored route from control board.";
221  nextParseState = ERROR_REPORT;
222  }
223  }
224  }
225  break;
226  // *********************************************************************
227  // Parse lat, lon and heading from periodic run of requestPosition()
228  // *********************************************************************
229  case POSITION_PARSE_DATA1:
230  {
231  QStringList positionList = incomingMessage.split(',');
232  if ((positionList.size() == 3) && (gpsState > 1))
233  {
234  float lat = positionList.at(0).toFloat() / 1.0E7;
235  float lon = positionList.at(1).toFloat() / 1.0E7;
236  if ((lat == 0.0) && (lon == 0.0))
237  return;
238  heading = positionList.at(2).toFloat() * 57.2957795;
239  vehicle->updatePosition(GeoDataCoordinates(lon, lat, 1000, GeoDataCoordinates::Degree));
240  vehicle->updateHeading(heading);
241  refreshMap(GeoDataCoordinates(lon, lat, 1000, GeoDataCoordinates::Degree));
242  }
243  }
244  break;
245  // *********************************************************************
246  // Parse altitude, course and speed from periodic run of requestPosition()
247  // *********************************************************************
248  case POSITION_PARSE_DATA2:
249  {
250  QStringList positionList = incomingMessage.split(',');
251  positionList[2].chop(1);
252  if (positionList.size() == 3)
253  {
254  //qDebug() << "Alt/Course/Speed: " << positionList[0] << positionList[1] << positionList [2];
255  }
256  }
257  break;
258  // *********************************************************************
259  // Parse # sats, accuracy, GPS state from periodic run of requestPosition()
260  // *********************************************************************
261  case POSITION_PARSE_SAT:
262  {
263  QStringList positionList = incomingMessage.split(',');
264  positionList[2].chop(1);
265  if (positionList.size() == 3)
266  {
267  //qDebug() << "Sats/Accuracy/State: " << positionList[0] << positionList[1] << positionList [2];
268  ui->sats->setText(positionList[0]);
269  gpsState = positionList[2].toInt();
270  if (gpsState == 0) ui->gpsState->setText("Detecting GPS");
271  if (gpsState == 1) ui->gpsState->setText("Waiting GPS Fix");
272  if (gpsState == 2) ui->gpsState->setText("2D GPS Fix");
273  if (gpsState == 3) ui->gpsState->setText("3D GPS Fix");
274  if (gpsState == 4) ui->gpsState->setText("3D Diff GPS Fix");
275  };
276  }
277  break;
278  // *********************************************************************
279  // This state gets executed when on_upload_clicked() is selected
280  // and starts the route upload process to the flight control board.
281  // *********************************************************************
282  case UPLOAD_WAYPOINT_COUNT:
283  // Check if waypoint count uploaded correctly
284  if (incomingMessage.toInt() == route->getRouteSize())
285  {
286  retryMessage = 0;
287  waypointIndex = 0;
288  uploadWaypoint(waypointIndex);
289  sendMessage("o0;");
290  nextParseState = UPLOAD_WAYPOINT;
291  }
292  else
293  {
294  retryMessage++;
295  if (retryMessage < MAX_MSG_RETRY)
296  {
297  sendMessage("o-1;");
298  nextParseState = UPLOAD_WAYPOINT_COUNT;
299  }
300  else
301  {
302  errorMessage = "Error uploading route, please try again";
303  nextParseState = ERROR_REPORT;
304  }
305  }
306  break;
307  // *********************************************************************
308  // This state verifies the current waypoint is successfully uploaded.
309  // After verification, the next waypoint is uploaded until route finished.
310  // *********************************************************************
311  case UPLOAD_WAYPOINT:
312  {
313  // Verify current waypoint loaded correctly
314  bool verifyWaypoint = false;
315  double waypointTolerance;
316  GeoDataCoordinates waypoint;
317  QStringList waypointQuery = incomingMessage.split(',');
318  if (waypointQuery.size() == 4) // check if correct number of parameters received
319  {
320  // There will be a little rounding error so calculate the difference for verification
321  waypoint = route->getWaypoint(waypointIndex);
322  waypointTolerance = fabs(waypoint.latitude(GeoDataCoordinates::Degree) - waypointQuery[1].toDouble()/1.0E7);
323  qDebug() << "check lat index[" + QString::number(waypointIndex) +"]" << waypointTolerance;
324  verifyWaypoint = waypointTolerance < WAYPOINT_ROUNDING_ERROR;
325  waypointTolerance = fabs(waypoint.longitude(GeoDataCoordinates::Degree) - waypointQuery[2].toDouble()/1.0E7);
326  qDebug() << "check lon index[" + QString::number(waypointIndex) +"]" << waypointTolerance;
327  verifyWaypoint &= waypointTolerance < WAYPOINT_ROUNDING_ERROR;
328  waypointTolerance = fabs(waypoint.altitude() - waypointQuery[3].toDouble()/1.0E7);
329  qDebug() << "check alt index[" + QString::number(waypointIndex) +"]" << waypointTolerance;
330  verifyWaypoint &= waypointTolerance < WAYPOINT_ROUNDING_ERROR;
331  }
332  char label = 'A';
333  emit panelStatus("Uploading waypoint " + QString(char(label+waypointIndex)) + "...");
334 
335 
336  // If current waypoint matches waypoint on flight control board, load next waypoint
337  if (verifyWaypoint)
338  {
339  waypointIndex++;
340  if (waypointIndex < route->getRouteSize())
341  {
342  retryMessage = 0;
343  uploadWaypoint(waypointIndex);
344  sendMessage("o" + QString::number(waypointIndex) + ";");
345  nextParseState = UPLOAD_WAYPOINT;
346  }
347  else // Finished uploading route, request position data for map
348  {
349  sendMessage("W"); // store route in EEPROM
350  QMessageBox::critical(this, "Route Upload", "Route uploaded successfully!", QMessageBox::Ok);
351  panelStatus("Waypoint upload successful.");
352  startPositionRequest();
353  }
354  }
355  // Current waypoint values does not match value on flight control board, resend current waypoint
356  else
357  {
358  retryMessage++;
359  if (retryMessage<MAX_MSG_RETRY)
360  {
361  waypoint = route->getWaypoint(waypointIndex);
362  QString command = "O" + QString::number(waypointIndex) + ";";
363  int coordinate = waypoint.latitude(GeoDataCoordinates::Degree)*1E7;
364  command += QString::number(coordinate) + ";";
365  coordinate = waypoint.longitude(GeoDataCoordinates::Degree)*1E7;
366  command += QString::number(coordinate) + ";";
367  command += QString::number(waypoint.altitude()) + ";";
368  sendMessage(command);
369  // Read back current waypoint from board
370  sendMessage("o" + QString::number(waypointIndex) + ";");
371  }
372  else
373  {
374  errorMessage = "Unable to upload route.\nPlease try again.";
375  panelStatus("Waypoint upload failed.");
376  nextParseState = ERROR_REPORT;
377  }
378  }
379  }
380  break;
381  case DEBUG_MSG:
382  {
383  data.chop(1);
384  qDebug() << "\nmessage size: " << data.size();
385  qDebug() << data.toHex();
386 // QByteArray floatData = data.mid(4, 4);
387 // QByteArray longData = parseData.mid(4, 4);
388 // QDataStream convertFloat(data.mid(0, 4));
389 // convertFloat.setByteOrder(QDataStream::LittleEndian);
390 // float floatNumber;
391 // convertFloat >> floatNumber;
392 // floatNumber = *reinterpret_cast<const float*>(floatData.data());
393 // qDebug() << "raw values: " << floatData << "converted float: " << floatNumber << "\n";
394  startPositionRequest();
395  }
396  break;
397  case UPDATE_AUTOPILOT_STATE:
398  {
399  QStringList state = incomingMessage.split(',');
400  if (state.size() == 1)
401  {
402  autoPilotStateChanged = false;
403  int type = incomingMessage.toInt();
404  //qDebug() << "Autopilot state response: " << type;
405  QString state;
406 
407  autoPilotState = false;
408  positionHoldState = false;
409  rtmState = false;
410 
411  if (type == AUTO_NAVIGATION)
412  autoPilotState = true;
413  if (autoPilotState)
414  state = "Autopilot\nEnabled";
415  else
416  state = "Autopilot\nDisabled";
417  updateIndicatorStatus(ui->autopilotState, autoPilotState, state);
418 
419  if (type == POSITION_HOLD)
420  positionHoldState = true;
421  if (positionHoldState)
422  state = "Position Hold\nEnabled";
423  else
424  state = "Position Hold\nDisabled";
425  updateIndicatorStatus(ui->positionHoldState, positionHoldState, state);
426 
427  if (type == RETURN_TO_HOME)
428  rtmState = true;
429  if (rtmState)
430  state = "Return\nTo Home\nEnabled";
431  else
432  state = "Return\nTo Home\nDisabled";
433  updateIndicatorStatus(ui->rtmState, rtmState, state);
434  }
435 
436  if (autoPilotStateChanged)
437  sendMessage("<");
438  else
439  startPositionRequest();
440  }
441  }
442  // *********************************************************************
443  // This state displays a dialog box indicating an error and begins
444  // requesting position reports for the map.
445  // *********************************************************************
446  if (nextParseState == ERROR_REPORT)
447  {
448  QMessageBox::critical(this, "Communicator", errorMessage, QMessageBox::Ok);
449  startPositionRequest();
450  }
