154  QString messages = data;
155  messages.chop(1); // remove LF
156  QList<QString> message = messages.split(',');
157  if (message.last() == "")
158  message.removeLast();

160  {
161  for (int index=0; index < plotSize; index++)
162  {

164  {
165  double dataValue = message[index].toDouble();
166  ui->plotWindow->graph(index)->addData(key, dataValue);
167  ui->plotWindow->graph(index)->removeDataBefore(key-timeAxisLength);
168  }
169  }
170  // make key axis range scroll with the data (at a constant range size of 10)
171  ui->plotWindow->xAxis->setRange(key+0.25, timeAxisLength, Qt::AlignRight);
172  ui->plotWindow->replot();

174  }
175  else if (message.size() > plotSize)
176  emit panelStatus("Plot count (" + QString::number(message.size()) +") does not match plot count in XML configuration file (" + QString::number(plotSize) +")");
