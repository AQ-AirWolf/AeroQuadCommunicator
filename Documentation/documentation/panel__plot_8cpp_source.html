48 void PanelPlot::initialize(QString filename)
49 {
50  // Open XML configuration file
51  QFile* file = new QFile(filename);
52  if (!file->open(QIODevice::ReadOnly | QIODevice::Text))
53  {
54  QMessageBox::critical(this, "AeroQuad Communicator", "Unable to find configuration file:\n" + QCoreApplication::applicationDirPath() + "/" + filename, QMessageBox::Ok);
55  file->close();
56  return;
57  }
58 
59  // Read XML file and populate data storage
60  plots.clear();
61  QXmlStreamReader xml(file);
62  while (!xml.atEnd() && !xml.hasError())
63  {
64  xml.readNext();
65  if (xml.isStartElement() && (xml.name() == "Plot"))
66  {
67  plot plotItem;
68  plotItem.name = xml.attributes().value("Name").toString();
69  xml.readNextStartElement();
70  while(!(xml.tokenType() == QXmlStreamReader::EndElement && xml.name() == "Plot"))
71  {
72  if(xml.tokenType() == QXmlStreamReader::StartElement)
73  {
74  if(xml.name() == "Requirement")
75  plotItem.requirement = xml.readElementText();
76  if(xml.name() == "Telemetry")
77  plotItem.telemetry = xml.readElementText();
78  if(xml.name() == "PlotName")
79  plotItem.plotNames.append(xml.readElementText());
80  }
81  xml.readNextStartElement();
82  }
83  plots.append(plotItem);
84  }
85  }
86  if (xml.hasError())
87  {
88  QMessageBox::warning(NULL,"Error",xml.errorString(),QMessageBox::Ok);
89  }
90  file->close();
91 }

140 void PanelPlot::displayCommData(QByteArray data)
141 {
142  double key = QDateTime::currentDateTime().toMSecsSinceEpoch()/1000.0;
143  QString messages = data;
144  if (messages == "initialize")
145  {
146  initializePlot(0);
147  setupPlotNames(0);
148  return;
149  }
150  messages.chop(2); // remove CR/LF
151  QList<QString> message = messages.split(',');
152  if (message.last() == "")
153  message.removeLast();
154  if (message.size() == plotSize)
155  {
156  for (int index=0; index < plotSize; index++)
157  {
158  if (ui->plotNames->item(index)->checkState() == Qt::Checked)
159  {
160  double dataValue = message[index].toDouble();
161  ui->plotWindow->graph(index)->addData(key, dataValue);
162  ui->plotWindow->graph(index)->removeDataBefore(key-timeAxisLength);
163  }
164  }
165  // make key axis range scroll with the data (at a constant range size of 10)
166  ui->plotWindow->xAxis->setRange(key+0.25, timeAxisLength, Qt::AlignRight);
167  ui->plotWindow->replot();
168  ui->plotWindow->rescaleAxes(true);
169  }
170  else if (message.size() > plotSize)
171  emit panelStatus("Plot count (" + QString::number(message.size()) +") does not match plot count in XML configuration file (" + QString::number(plotSize) +")");
172 }
